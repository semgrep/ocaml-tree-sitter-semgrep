    V1(u8, bool),
    V2(S3),

    x.0.$FIELD.x + x.1.x + x.2.0.x;
    x.0.0.x = 0;
    x.1.x = 1;
    x.2.0.x = 2;
}

fun assign_enum(x: &mut E) {
    match (x) {
        E::V1($VAR, y, ...) => {
            *$VAR = 42;
            *y = true;

        E::V2(x) => {
            x.0.0.x = 0;
            x.1.x = 1;
            x.2.0.x = 2;
        },
        ...

}
